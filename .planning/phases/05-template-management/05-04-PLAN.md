"""
Template preview generation for Vakalat AI.

Provides lightweight preview of templates showing required inputs
without including full template content.
"""

from typing import List, Optional
from pydantic import BaseModel, Field

from src.templates.schemas import (
    LegalTemplate,
    TemplateField,
    TemplateStatus,
    DocumentType,
    CourtLevel,
)
from src.templates.storage import TemplateRepository


class FieldPreview(BaseModel):
    """Preview of a single template field."""
    field_name: str = Field(description="Unique field identifier")
    label: str = Field(description="Human-readable field label")
    field_type: str = Field(description="Field data type (text, date, party, etc.)")
    required: bool = Field(description="Whether field is required")
    description: str = Field(description="What this field contains")
    example: Optional[str] = Field(default=None, description="Example value")
    validation_regex: Optional[str] = Field(default=None, description="Validation pattern if any")


class TemplatePreview(BaseModel):
    """
    Lightweight preview of a template.

    Shows metadata and field requirements without full template content.
    Used for template selection UI before document generation.
    """
    # Template identification
    doc_type: str = Field(description="Document type")
    court_level: str = Field(description="Court level")
    name: str = Field(description="Template name")
    description: str = Field(description="Template description")

    # Version and status
    version: str = Field(description="Current template version")
    status: str = Field(description="Lifecycle status (active/deprecated/archived)")

    # Field counts for quick reference
    required_field_count: int = Field(description="Number of required fields")
    optional_field_count: int = Field(description="Number of optional fields")

    # Field details
    required_fields: List[FieldPreview] = Field(description="Required input fields")
    optional_fields: List[FieldPreview] = Field(description="Optional input fields")

    # Formatting summary
    font: str = Field(description="Document font")
    font_size: int = Field(description="Document font size in points")


def _field_to_preview(field: TemplateField, required: bool) -> FieldPreview:
    """Convert a TemplateField to FieldPreview."""
    return FieldPreview(
        field_name=field.field_name,
        label=field.label,
        field_type=field.field_type,
        required=required,
        description=field.description,
        example=field.example,
        validation_regex=field.validation_regex
    )


def generate_preview(template: LegalTemplate) -> TemplatePreview:
    """
    Generate a lightweight preview from a full template.

    Args:
        template: Full LegalTemplate object

    Returns:
        TemplatePreview with field metadata
    """
    return TemplatePreview(
        doc_type=template.metadata.doc_type.value,
        court_level=template.metadata.court_level.value,
        name=template.metadata.name,
        description=template.metadata.description,
        version=template.metadata.version,
        status=template.status.value,
        required_field_count=len(template.required_fields),
        optional_field_count=len(template.optional_fields),
        required_fields=[
            _field_to_preview(f, required=True)
            for f in template.required_fields
        ],
        optional_fields=[
            _field_to_preview(f, required=False)
            for f in template.optional_fields
        ],
        font=template.formatting.font,
        font_size=template.formatting.font_size
    )


def get_template_preview(
    doc_type: DocumentType,
    court_level: CourtLevel,
    repo: Optional[TemplateRepository] = None
) -> Optional[TemplatePreview]:
    """
    Get preview for a specific template.

    Args:
        doc_type: Document type
        court_level: Court level
        repo: Template repository

    Returns:
        TemplatePreview or None if template not found
    """
    if repo is None:
        repo = TemplateRepository()

    template = repo.get_template(doc_type, court_level)
    if template is None:
        return None

    return generate_preview(template)


def list_template_previews(
    doc_type: Optional[DocumentType] = None,
    repo: Optional[TemplateRepository] = None,
    include_deprecated: bool = True,
    include_archived: bool = False
) -> List[TemplatePreview]:
    """
    List previews for all templates.

    Args:
        doc_type: Optional filter by document type
        repo: Template repository
        include_deprecated: Include deprecated templates (default True)
        include_archived: Include archived templates (default False)

    Returns:
        List of TemplatePreview objects
    """
    if repo is None:
        repo = TemplateRepository()

    templates = repo.list_templates(doc_type)
    previews = []

    for template in templates:
        # Filter by status
        if template.status == TemplateStatus.DEPRECATED and not include_deprecated:
            continue
        if template.status == TemplateStatus.ARCHIVED and not include_archived:
            continue

        previews.append(generate_preview(template))

    return previews
