"""
Template lifecycle status management.

Handles template status transitions (active -> deprecated -> archived)
with validation and changelog tracking.
"""

import logging
from dataclasses import dataclass
from typing import Optional, Tuple
from datetime import datetime

from src.templates.schemas import (
    LegalTemplate,
    TemplateStatus,
    ChangelogEntry,
    ALLOWED_TRANSITIONS,
    DocumentType,
    CourtLevel,
)
from src.templates.storage import TemplateRepository


logger = logging.getLogger(__name__)


@dataclass
class StatusChangeResult:
    """Result of a template status change operation."""
    success: bool
    old_status: Optional[TemplateStatus] = None
    new_status: Optional[TemplateStatus] = None
    template_name: Optional[str] = None
    error_message: Optional[str] = None


def validate_status_transition(
    current_status: TemplateStatus,
    new_status: TemplateStatus
) -> Tuple[bool, Optional[str]]:
    """
    Validate that a status transition is allowed.

    Args:
        current_status: Current template status
        new_status: Proposed new status

    Returns:
        Tuple of (is_valid, error_message)
    """
    # Same status is always valid (no-op)
    if current_status == new_status:
        return True, None

    allowed = ALLOWED_TRANSITIONS.get(current_status, set())

    if new_status not in allowed:
        allowed_values = [s.value for s in allowed] if allowed else []
        if not allowed_values:
            return False, f"Cannot change status from {current_status.value} (terminal state)"
        return False, (
            f"Cannot transition from {current_status.value} to {new_status.value}. "
            f"Allowed transitions: {allowed_values}"
        )

    return True, None


def change_template_status(
    doc_type: DocumentType,
    court_level: CourtLevel,
    new_status: TemplateStatus,
    reason: str,
    repo: Optional[TemplateRepository] = None,
    author: str = "system"
) -> StatusChangeResult:
    """
    Change a template's lifecycle status.

    Valid transitions:
    - active -> deprecated (template still usable with warning)
    - active -> archived (template no longer usable)
    - deprecated -> archived (template no longer usable)

    Archived is a terminal state - cannot be reversed.

    Args:
        doc_type: Document type of template
        court_level: Court level of template
        new_status: New status to set
        reason: Reason for status change (for changelog)
        repo: Template repository (uses default if not provided)
        author: Author name for changelog

    Returns:
        StatusChangeResult with success status and details
    """
    if repo is None:
        repo = TemplateRepository()

    # Load template
    template = repo.get_template(doc_type, court_level)
    if template is None:
        return StatusChangeResult(
            success=False,
            error_message=f"Template not found: {doc_type.value}/{court_level.value}"
        )

    current_status = template.status

    # Validate transition
    is_valid, error = validate_status_transition(current_status, new_status)
    if not is_valid:
        return StatusChangeResult(
            success=False,
            old_status=current_status,
            new_status=new_status,
            error_message=error
        )

    # No-op if same status
    if current_status == new_status:
        return StatusChangeResult(
            success=True,
            old_status=current_status,
            new_status=new_status,
            template_name=template.metadata.name
        )

    # Create changelog entry for status change
    change_entry = ChangelogEntry(
        version=template.metadata.version,
        date=datetime.now().isoformat(),
        changes=[f"Status changed from {current_status.value} to {new_status.value}: {reason}"],
        author=author
    )

    # Update template with new status and changelog
    existing_changelog = list(template.changelog) if template.changelog else []
    updated_template = template.model_copy(update={
        "status": new_status,
        "changelog": [change_entry] + existing_changelog,
        "updated_at": datetime.now().isoformat()
    })

    # Save updated template
    try:
        repo.save_template(updated_template)
        logger.info(
            f"Template status changed: {doc_type.value}/{court_level.value} "
            f"{current_status.value} -> {new_status.value}"
        )
    except Exception as e:
        logger.error(f"Failed to save template status change: {e}")
        return StatusChangeResult(
            success=False,
            old_status=current_status,
            new_status=new_status,
            error_message=f"Failed to save: {str(e)}"
        )

    return StatusChangeResult(
        success=True,
        old_status=current_status,
        new_status=new_status,
        template_name=template.metadata.name
    )


def get_template_status(
    doc_type: DocumentType,
    court_level: CourtLevel,
    repo: Optional[TemplateRepository] = None
) -> Optional[TemplateStatus]:
    """
    Get current status of a template.

    Args:
        doc_type: Document type
        court_level: Court level
        repo: Template repository

    Returns:
        TemplateStatus or None if template not found
    """
    if repo is None:
        repo = TemplateRepository()

    template = repo.get_template(doc_type, court_level)
    if template is None:
        return None

    return template.status


def is_template_usable(
    doc_type: DocumentType,
    court_level: CourtLevel,
    repo: Optional[TemplateRepository] = None
) -> Tuple[bool, Optional[str]]:
    """
    Check if a template can be used for document generation.

    Deprecated templates are usable but return a warning.
    Archived templates are not usable.

    Args:
        doc_type: Document type
        court_level: Court level
        repo: Template repository

    Returns:
        Tuple of (is_usable, warning_message)
        - (True, None) for active templates
        - (True, "warning...") for deprecated templates
        - (False, "error...") for archived templates
    """
    if repo is None:
        repo = TemplateRepository()

    template = repo.get_template(doc_type, court_level)
    if template is None:
        return False, f"Template not found: {doc_type.value}/{court_level.value}"

    if template.status == TemplateStatus.ACTIVE:
        return True, None
    elif template.status == TemplateStatus.DEPRECATED:
        return True, (
            f"Warning: Template '{template.metadata.name}' is deprecated. "
            "Consider using an alternative template."
        )
    else:  # ARCHIVED
        return False, (
            f"Template '{template.metadata.name}' is archived and cannot be used. "
            "Please select an active template."
        )
