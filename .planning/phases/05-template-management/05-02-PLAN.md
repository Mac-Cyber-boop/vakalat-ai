```python
"""
Version management utilities for legal templates.

Provides semantic versioning comparison and changelog management
for template version tracking.
"""

import semver
from datetime import datetime
from typing import List, Optional, Tuple

from src.templates.schemas import ChangelogEntry


def compare_versions(version_a: str, version_b: str) -> int:
    """
    Compare two semantic versions.

    Args:
        version_a: First version string (e.g., "1.0.0")
        version_b: Second version string (e.g., "1.1.0")

    Returns:
        -1 if version_a < version_b
         0 if version_a == version_b
         1 if version_a > version_b

    Raises:
        ValueError: If either version is not valid semver
    """
    try:
        ver_a = semver.Version.parse(version_a)
        ver_b = semver.Version.parse(version_b)
    except ValueError as e:
        raise ValueError(f"Invalid semantic version: {e}")

    return ver_a.compare(ver_b)


def is_version_higher(new_version: str, existing_version: str) -> bool:
    """
    Check if new version is higher than existing version.

    Args:
        new_version: Proposed new version
        existing_version: Current template version

    Returns:
        True if new_version > existing_version
    """
    return compare_versions(new_version, existing_version) > 0


def bump_version(
    current_version: str,
    bump_type: str = "patch"
) -> str:
    """
    Bump a semantic version.

    Args:
        current_version: Current version string
        bump_type: "major", "minor", or "patch"

    Returns:
        New version string

    Raises:
        ValueError: If bump_type is invalid
    """
    ver = semver.Version.parse(current_version)

    if bump_type == "major":
        return str(ver.bump_major())
    elif bump_type == "minor":
        return str(ver.bump_minor())
    elif bump_type == "patch":
        return str(ver.bump_patch())
    else:
        raise ValueError(f"bump_type must be 'major', 'minor', or 'patch', got '{bump_type}'")


def create_changelog_entry(
    version: str,
    changes: List[str],
    author: str = "system"
) -> ChangelogEntry:
    """
    Create a new changelog entry.

    Args:
        version: Version this entry is for
        changes: List of change descriptions
        author: Who made the changes

    Returns:
        ChangelogEntry with current timestamp
    """
    return ChangelogEntry(
        version=version,
        date=datetime.now().isoformat(),
        changes=changes,
        author=author
    )


def validate_version_upgrade(
    new_version: str,
    existing_version: Optional[str]
) -> Tuple[bool, Optional[str]]:
    """
    Validate that a version upgrade is valid.

    Args:
        new_version: Proposed new version
        existing_version: Current version (None if new template)

    Returns:
        Tuple of (is_valid, error_message)
        - (True, None) if valid
        - (False, "error message") if invalid
    """
    # Validate new version is valid semver
    try:
        semver.Version.parse(new_version)
    except ValueError:
        return False, f"Invalid version format: '{new_version}'. Use semantic versioning (e.g., '1.0.0')"

    # If no existing version, new version just needs to be valid
    if existing_version is None:
        return True, None

    # Validate existing version
    try:
        semver.Version.parse(existing_version)
    except ValueError:
        # Existing version invalid - allow any valid new version
        return True, None

    # Check new is higher than existing
    if not is_version_higher(new_version, existing_version):
        return False, f"Version {new_version} must be higher than existing version {existing_version}"

    return True, None
```

```python
"""
Template upload validation and processing.

Handles JSON file upload validation against LegalTemplate schema,
with detailed error messages for validation failures.
"""

import json
import logging
from typing import Tuple, Optional, Dict, Any, List
from dataclasses import dataclass

from pydantic import ValidationError

from src.templates.schemas import LegalTemplate, TemplateStatus
from src.templates.storage import TemplateRepository
from src.templates.versioning import validate_version_upgrade, create_changelog_entry


logger = logging.getLogger(__name__)

# Maximum file size for template uploads (1MB)
MAX_UPLOAD_SIZE = 1024 * 1024


@dataclass
class ValidationResult:
    """Result of template validation."""
    is_valid: bool
    template: Optional[LegalTemplate] = None
    errors: Optional[List[Dict[str, str]]] = None
    error_message: Optional[str] = None


@dataclass
class UploadResult:
    """Result of template upload processing."""
    success: bool
    is_update: bool = False
    template_name: Optional[str] = None
    doc_type: Optional[str] = None
    court_level: Optional[str] = None
    version: Optional[str] = None
    error_message: Optional[str] = None
    validation_errors: Optional[List[Dict[str, str]]] = None


def validate_json_syntax(content: bytes) -> Tuple[bool, Optional[str], Optional[dict]]:
    """
    Validate JSON syntax.

    Args:
        content: Raw bytes content

    Returns:
        Tuple of (is_valid, error_message, parsed_data)
    """
    try:
        data = json.loads(content.decode('utf-8'))
        return True, None, data
    except json.JSONDecodeError as e:
        return False, f"Invalid JSON syntax at line {e.lineno}: {e.msg}", None
    except UnicodeDecodeError as e:
        return False, f"File must be UTF-8 encoded: {e}", None


def validate_template_schema(content: bytes) -> ValidationResult:
    """
    Validate template JSON against LegalTemplate schema.

    Args:
        content: JSON content as bytes

    Returns:
        ValidationResult with template or error details
    """
    # Check JSON syntax first
    is_valid, error, _ = validate_json_syntax(content)
    if not is_valid:
        return ValidationResult(
            is_valid=False,
            error_message=error
        )

    # Validate against Pydantic schema
    try:
        template = LegalTemplate.model_validate_json(content)
        return ValidationResult(
            is_valid=True,
            template=template
        )
    except ValidationError as e:
        errors = [
            {
                "field": ".".join(map(str, err["loc"])),
                "message": err["msg"],
                "type": err["type"]
            }
            for err in e.errors()
        ]
        return ValidationResult(
            is_valid=False,
            errors=errors,
            error_message=f"Template validation failed with {len(errors)} error(s)"
        )


def validate_template_upload(
    content: bytes,
    content_type: Optional[str] = None,
    repo: Optional[TemplateRepository] = None
) -> ValidationResult:
    """
    Full validation of template upload including version check.

    Args:
        content: File content as bytes
        content_type: MIME type of uploaded file
        repo: Template repository for version checking

    Returns:
        ValidationResult with validated template or errors
    """
    # Check content type if provided
    if content_type and "json" not in content_type.lower():
        return ValidationResult(
            is_valid=False,
            error_message="File must be JSON (application/json)"
        )

    # Check file size
    if len(content) > MAX_UPLOAD_SIZE:
        return ValidationResult(
            is_valid=False,
            error_message=f"File too large. Maximum size: {MAX_UPLOAD_SIZE} bytes"
        )

    # Validate schema
    result = validate_template_schema(content)
    if not result.is_valid:
        return result

    template = result.template

    # Check version against existing if repo provided
    if repo:
        existing = repo.get_template(
            template.metadata.doc_type,
            template.metadata.court_level
        )
        if existing:
            existing_version = existing.metadata.version
            new_version = template.metadata.version

            is_valid, error = validate_version_upgrade(new_version, existing_version)
            if not is_valid:
                return ValidationResult(
                    is_valid=False,
                    error_message=error
                )

    return ValidationResult(
        is_valid=True,
        template=template
    )


def process_template_upload(
    content: bytes,
    content_type: Optional[str] = None,
    repo: Optional[TemplateRepository] = None,
    change_description: Optional[List[str]] = None,
    author: str = "user"
) -> UploadResult:
    """
    Process and save an uploaded template.

    Args:
        content: File content as bytes
        content_type: MIME type of uploaded file
        repo: Template repository
        change_description: Description of changes for changelog
        author: Author name for changelog

    Returns:
        UploadResult with success status and details
    """
    if repo is None:
        repo = TemplateRepository()

    # Validate upload
    validation = validate_template_upload(content, content_type, repo)
    if not validation.is_valid:
        return UploadResult(
            success=False,
            error_message=validation.error_message,
            validation_errors=validation.errors
        )

    template = validation.template

    # Check if this is an update
    existing = repo.get_template(
        template.metadata.doc_type,
        template.metadata.court_level
    )
    is_update = existing is not None

    # Add changelog entry if this is an update
    if is_update and change_description:
        entry = create_changelog_entry(
            version=template.metadata.version,
            changes=change_description,
            author=author
        )
        # Prepend new entry to changelog (preserve old entries)
        existing_changelog = list(template.changelog) if template.changelog else []
        if existing:
            existing_changelog = list(existing.changelog) if existing.changelog else []
        template = template.model_copy(update={
            "changelog": [entry] + existing_changelog
        })
    elif not is_update:
        # First version - create initial changelog entry
        entry = create_changelog_entry(
            version=template.metadata.version,
            changes=change_description or ["Initial template creation"],
            author=author
        )
        template = template.model_copy(update={
            "changelog": [entry]
        })

    # Save template
    try:
        repo.save_template(template)
        logger.info(
            f"Template uploaded: {template.metadata.doc_type.value}/"
            f"{template.metadata.court_level.value} v{template.metadata.version}"
        )
    except Exception as e:
        logger.error(f"Failed to save template: {e}")
        return UploadResult(
            success=False,
            error_message=f"Failed to save template: {str(e)}"
        )

    return UploadResult(
        success=True,
        is_update=is_update,
        template_name=template.metadata.name,
        doc_type=template.metadata.doc_type.value,
        court_level=template.metadata.court_level.value,
        version=template.metadata.version
    )
```
