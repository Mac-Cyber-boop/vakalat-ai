# 12. TEMPLATE MANAGEMENT ENDPOINTS

@app.post("/templates/upload")
async def upload_template(file: UploadFile):
    """
    Upload and validate a custom template.

    Accepts a JSON file containing a LegalTemplate. Validates:
    - JSON syntax
    - Schema compliance (all required fields present)
    - Version is valid semver
    - Version is higher than existing (if updating)

    Returns validation results with specific error messages if invalid.
    """
    # Validate content type
    if not file.content_type or "json" not in file.content_type.lower():
        raise HTTPException(
            status_code=400,
            detail="File must be JSON (application/json)"
        )

    # Read file contents
    contents = await file.read()

    # Check file size
    if len(contents) > MAX_UPLOAD_SIZE:
        raise HTTPException(
            status_code=400,
            detail=f"File too large. Maximum size: {MAX_UPLOAD_SIZE} bytes"
        )

    # Process upload
    result = process_template_upload(
        content=contents,
        content_type=file.content_type,
        repo=template_repo,
        change_description=["Uploaded via API"],
        author="api"
    )

    if not result.success:
        detail = {"message": result.error_message}
        if result.validation_errors:
            detail["validation_errors"] = result.validation_errors
        raise HTTPException(status_code=400, detail=detail)

    return {
        "status": "success",
        "message": "Template uploaded successfully",
        "template": {
            "doc_type": result.doc_type,
            "court_level": result.court_level,
            "version": result.version,
            "name": result.template_name
        },
        "is_update": result.is_update
    }


@app.post("/templates/history")
async def get_template_history(req: TemplateHistoryRequest):
    """
    Get version history with changelog for a template.

    Returns the changelog entries showing version progression,
    dates of changes, and descriptions of what changed.
    """
    # Validate doc_type
    try:
        doc_type = DocumentType(req.doc_type)
    except ValueError:
        valid_types = [dt.value for dt in DocumentType]
        raise HTTPException(
            400,
            f"Invalid doc_type '{req.doc_type}'. Valid types: {valid_types}"
        )

    # Validate court_level
    try:
        court_level = CourtLevel(req.court_level)
    except ValueError:
        valid_levels = [cl.value for cl in CourtLevel]
        raise HTTPException(
            400,
            f"Invalid court_level '{req.court_level}'. Valid levels: {valid_levels}"
        )

    template = template_repo.get_template(doc_type, court_level)
    if template is None:
        raise HTTPException(
            404,
            f"Template not found: {req.doc_type}/{req.court_level}"
        )

    changelog = template.changelog if template.changelog else []

    return {
        "doc_type": req.doc_type,
        "court_level": req.court_level,
        "current_version": template.metadata.version,
        "current_status": template.status.value,
        "history": [
            {
                "version": entry.version,
                "date": entry.date,
                "changes": entry.changes,
                "author": entry.author
            }
            for entry in changelog
        ]
    }


@app.post("/templates/status")
async def change_status(req: TemplateStatusChangeRequest):
    """
    Change template lifecycle status.

    Valid transitions:
    - active -> deprecated (template still usable with warning)
    - active -> archived (template no longer usable)
    - deprecated -> archived (template no longer usable)

    Archived is a terminal state and cannot be reversed.
    Reason for change is recorded in changelog.
    """
    # Validate doc_type
    try:
        doc_type = DocumentType(req.doc_type)
    except ValueError:
        valid_types = [dt.value for dt in DocumentType]
        raise HTTPException(
            400,
            f"Invalid doc_type '{req.doc_type}'. Valid types: {valid_types}"
        )

    # Validate court_level
    try:
        court_level = CourtLevel(req.court_level)
    except ValueError:
        valid_levels = [cl.value for cl in CourtLevel]
        raise HTTPException(
            400,
            f"Invalid court_level '{req.court_level}'. Valid levels: {valid_levels}"
        )

    # Validate new_status
    try:
        new_status = TemplateStatus(req.new_status)
    except ValueError:
        valid_statuses = [s.value for s in TemplateStatus]
        raise HTTPException(
            400,
            f"Invalid status '{req.new_status}'. Valid statuses: {valid_statuses}"
        )

    # Execute status change
    result = change_template_status(
        doc_type=doc_type,
        court_level=court_level,
        new_status=new_status,
        reason=req.reason,
        repo=template_repo,
        author="api"
    )

    if not result.success:
        raise HTTPException(400, result.error_message)

    return {
        "status": "success",
        "template_name": result.template_name,
        "old_status": result.old_status.value if result.old_status else None,
        "new_status": result.new_status.value if result.new_status else None,
        "reason": req.reason
    }


@app.post("/templates/preview")
async def preview_template(req: TemplatePreviewRequest):
    """
    Preview a template showing required input fields.

    Returns a lightweight preview with:
    - Template metadata (name, version, status)
    - List of required fields with labels and types
    - List of optional fields
    - Formatting information (font, size)

    Does NOT return the full template_content.
    Useful for template selection before document generation.
    """
    # Validate doc_type
    try:
        doc_type = DocumentType(req.doc_type)
    except ValueError:
        valid_types = [dt.value for dt in DocumentType]
        raise HTTPException(
            400,
            f"Invalid doc_type '{req.doc_type}'. Valid types: {valid_types}"
        )

    # Validate court_level
    try:
        court_level = CourtLevel(req.court_level)
    except ValueError:
        valid_levels = [cl.value for cl in CourtLevel]
        raise HTTPException(
            400,
            f"Invalid court_level '{req.court_level}'. Valid levels: {valid_levels}"
        )

    preview = get_template_preview(doc_type, court_level, template_repo)

    if preview is None:
        raise HTTPException(
            404,
            f"Template not found: {req.doc_type}/{req.court_level}"
        )

    # Check if deprecated (add warning)
    warning = None
    if preview.status == "deprecated":
        warning = f"Warning: Template '{preview.name}' is deprecated. Consider using an alternative."

    response = preview.model_dump()
    if warning:
        response["warning"] = warning

    return response
