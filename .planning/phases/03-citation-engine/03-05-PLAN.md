```python
def display_precedents(recommendations: list):
    """Display precedent recommendations with verification badges."""
    if not recommendations:
        st.info("No relevant precedents found for this query.")
        return

    st.subheader("Relevant Precedents")
    for rec in recommendations:
        status = rec.get("verification_status", "unverified")
        badge_class = "verified" if status == "verified" else "unverified"
        badge_icon = "&#10003;" if status == "verified" else "&#9888;"
        badge_title = "Verified in database" if status == "verified" else "Could not verify"

        html = f"""
        <div class="precedent-card {badge_class}">
            <strong>{rec.get('formatted_citation', rec.get('case_name', 'Unknown'))}</strong>
            <span class="citation-badge {badge_class}" title="{badge_title}">{badge_icon}</span>
            <br/>
            <small>Court: {rec.get('court', 'N/A')} | Year: {rec.get('year', 'N/A')} | Relevance: {rec.get('relevance_score', 0):.2f}</small>
            <p style="margin-top: 8px; font-size: 0.9em;">{rec.get('snippet', '')[:300]}...</p>
        </div>
        """
        st.markdown(html, unsafe_allow_html=True)
```

This function can be called when displaying research results that include precedents.
