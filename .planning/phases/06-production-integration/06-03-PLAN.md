# 13. EXPORT ENDPOINTS

@app.post("/export/pdf")
async def export_pdf(req: ExportDocumentRequest):
    """
    Export document content to PDF with court-standard formatting.

    Retrieves formatting requirements from the template for the specified
    doc_type and court_level, then generates a PDF with:
    - Court-specific margins (Supreme Court: 4cm, District Court: 3cm)
    - Times New Roman font at specified size
    - 1.5 line spacing
    - Automatic page numbers in footer

    Returns: Binary PDF file as attachment download
    """
    # Validate doc_type
    try:
        doc_type = DocumentType(req.doc_type)
    except ValueError:
        valid_types = [dt.value for dt in DocumentType]
        raise HTTPException(
            400,
            f"Invalid doc_type '{req.doc_type}'. Valid types: {valid_types}"
        )

    # Validate court_level
    try:
        court_level = CourtLevel(req.court_level)
    except ValueError:
        valid_levels = [cl.value for cl in CourtLevel]
        raise HTTPException(
            400,
            f"Invalid court_level '{req.court_level}'. Valid levels: {valid_levels}"
        )

    # Get formatting from template
    template = template_repo.get_template(doc_type, court_level)
    if template is None:
        raise HTTPException(
            404,
            f"Template not found for {req.doc_type} at {req.court_level}"
        )

    formatting = template.formatting

    # Generate PDF
    try:
        pdf_buffer = export_to_pdf(req.content, formatting)
    except Exception as e:
        raise HTTPException(500, f"PDF generation failed: {str(e)}")

    # Generate filename
    filename = req.filename or f"{req.doc_type}_{req.court_level}"

    return StreamingResponse(
        pdf_buffer,
        media_type="application/pdf",
        headers={
            "Content-Disposition": f"attachment; filename={filename}.pdf"
        }
    )
```

@app.post("/export/docx")
async def export_docx(req: ExportDocumentRequest):
    """
    Export document content to DOCX with court-standard formatting.

    Retrieves formatting requirements from the template for the specified
    doc_type and court_level, then generates a DOCX with:
    - Court-specific margins
    - Times New Roman font at specified size
    - Proper line spacing
    - A4 page size

    The DOCX can be edited in Microsoft Word while preserving formatting.

    Returns: Binary DOCX file as attachment download
    """
    # Validate doc_type
    try:
        doc_type = DocumentType(req.doc_type)
    except ValueError:
        valid_types = [dt.value for dt in DocumentType]
        raise HTTPException(
            400,
            f"Invalid doc_type '{req.doc_type}'. Valid types: {valid_types}"
        )

    # Validate court_level
    try:
        court_level = CourtLevel(req.court_level)
    except ValueError:
        valid_levels = [cl.value for cl in CourtLevel]
        raise HTTPException(
            400,
            f"Invalid court_level '{req.court_level}'. Valid levels: {valid_levels}"
        )

    # Get formatting from template
    template = template_repo.get_template(doc_type, court_level)
    if template is None:
        raise HTTPException(
            404,
            f"Template not found for {req.doc_type} at {req.court_level}"
        )

    formatting = template.formatting

    # Generate DOCX
    try:
        docx_buffer = export_to_docx(req.content, formatting)
    except Exception as e:
        raise HTTPException(500, f"DOCX generation failed: {str(e)}")

    # Generate filename
    filename = req.filename or f"{req.doc_type}_{req.court_level}"

    return StreamingResponse(
        docx_buffer,
        media_type="application/vnd.openxmlformats-officedocument.wordprocessingml.document",
        headers={
            "Content-Disposition": f"attachment; filename={filename}.docx"
        }
    )
```
