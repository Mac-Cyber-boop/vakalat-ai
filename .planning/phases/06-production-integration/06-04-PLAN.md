```python
# 14. PROFESSIONAL DRAFTING ENDPOINT

@app.post("/draft-pro")
async def draft_pro(req: DraftProRequest):
    """
    Professional document drafting with optional export.

    This is the unified endpoint for production document generation:
    1. Validates document type and court level
    2. Validates facts against document-specific schema
    3. Generates formal legal document with citation integration
    4. Verifies all citations before including
    5. Optionally exports to PDF or DOCX

    The endpoint integrates:
    - Template loading (Phase 2)
    - Citation recommendation (Phase 3)
    - Citation verification (Phase 1)
    - Document generation (Phase 4)
    - Export (Phase 6)

    Args:
        req: DraftProRequest with doc_type, court_level, facts, optional export_format

    Returns:
        - If export_format is None: JSON with DraftProResponse
        - If export_format is 'pdf': Binary PDF download
        - If export_format is 'docx': Binary DOCX download

    Raises:
        400: Invalid doc_type, court_level, facts, or export_format
        404: Template not found
        500: Generation or export failed
    """
    if not document_generator:
        raise HTTPException(500, "Document generator not available")

    # Validate doc_type
    try:
        doc_type = DocumentType(req.doc_type)
    except ValueError:
        valid_types = [dt.value for dt in DocumentType]
        raise HTTPException(
            400,
            f"Invalid doc_type '{req.doc_type}'. Valid types: {valid_types}"
        )

    # Validate court_level
    try:
        court_level = CourtLevel(req.court_level)
    except ValueError:
        valid_levels = [cl.value for cl in CourtLevel]
        raise HTTPException(
            400,
            f"Invalid court_level '{req.court_level}'. Valid levels: {valid_levels}"
        )

    # Validate export_format if provided
    if req.export_format and req.export_format not in ('pdf', 'docx'):
        raise HTTPException(
            400,
            f"Invalid export_format '{req.export_format}'. Valid formats: pdf, docx"
        )

    # Validate facts based on doc_type
    fact_model_map = {
        DocumentType.BAIL_APPLICATION: BailApplicationFacts,
        DocumentType.LEGAL_NOTICE: LegalNoticeFacts,
        DocumentType.AFFIDAVIT: AffidavitFacts,
        DocumentType.PETITION: PetitionFacts,
    }

    fact_model = fact_model_map.get(doc_type)
    if fact_model:
        try:
            validated_facts = fact_model(**req.facts)
            facts_dict = validated_facts.model_dump()
        except Exception as e:
            raise HTTPException(400, f"Invalid facts: {str(e)}")
    else:
        facts_dict = req.facts

    # Generate document
    try:
        result: GeneratedDocument = document_generator.generate_document(
            doc_type=doc_type,
            court_level=court_level,
            user_facts=facts_dict
        )
    except ValueError as e:
        raise HTTPException(400, str(e))
    except Exception as e:
        raise HTTPException(500, f"Generation failed: {str(e)}")

    # If no export requested, return JSON response
    if not req.export_format:
        return result.model_dump()

    # Get formatting from the generated document
    from src.templates.schemas import FormattingRequirements
    formatting = FormattingRequirements(**result.formatting)

    # Generate filename
    filename = req.filename or f"{req.doc_type}_{req.court_level}"

    # Export to requested format
    if req.export_format == 'pdf':
        try:
            pdf_buffer = export_to_pdf(result.content, formatting)
        except Exception as e:
            raise HTTPException(500, f"PDF export failed: {str(e)}")

        return StreamingResponse(
            pdf_buffer,
            media_type="application/pdf",
            headers={
                "Content-Disposition": f"attachment; filename={filename}.pdf"
            }
        )
    else:  # docx
        try:
            docx_buffer = export_to_docx(result.content, formatting)
        except Exception as e:
            raise HTTPException(500, f"DOCX export failed: {str(e)}")

        return StreamingResponse(
            docx_buffer,
            media_type="application/vnd.openxmlformats-officedocument.wordprocessingml.document",
            headers={
                "Content-Disposition": f"attachment; filename={filename}.docx"
            }
        )
```
